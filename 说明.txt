第三步：口语化写作，去营销腔
规则： AI偏好自然语言。写完后读出来——像正常说话吗？

代码：Python可读性检测脚本

import re
def check_readability(text):
    """检测文本是否适合AI引擎抓取"""
    sentences = re.split(r'[。！？\n]', text)
    sentences = [s.strip() for s in sentences if len(s.strip()) > 5]
    # 1. 营销词检测
    buzzwords = ['创新', '领先', '卓越', '全方位', '一站式', '极致', 
                 '颠覆', '最佳', '最优质', '革命性']
    buzz_hits = [w for w in buzzwords if w in text]
    # 2. 平均句子长度（AI偏好15-25字）
    avg_len = sum(len(s) for s in sentences) / len(sentences) if sentences else 0
    # 3. 是否有直接问句（AI友好信号）
    has_questions = any('?' in s or '？' in s for s in sentences)
    # 4. 是否含数据/数字
    numbers = re.findall(r'\d+', text)
    print("=== AI可读性检查报告 ===")
    print(f"总段落数: {len(sentences)}")
    print(f"平均句长: {avg_len:.0f}字 {'✅' if 15 <= avg_len <= 25 else '⚠️ 理想范围15-25字'}")
    print(f"营销词命中: {buzz_hits if buzz_hits else '无 ✅'}")
    print(f"包含问句: {'✅' if has_questions else '❌ 建议加入直接问句'}")
    print(f"数据引用: {len(numbers)}个数字 {'✅' if numbers else '❌ 建议加入具体数据'}")
    print(f"AI友好度: {'高' if avg_len <= 25 and has_questions and numbers else '中' if avg_len <= 35 else '低'}")





# 使用示例
text = """我们提供创新的全方位跨境解决方案，
通过领先的AI技术为客户创造卓越价值。"""
check_readability(text)

import requests
import json
from datetime import datetime
# 需要Perplexity API Key: https://docs.perplexity.ai/
PERPLEXITY_API_KEY = "your_key_here"
def test_ai_visibility(brand_name, keyword):
    """测试品牌关键词在Perplexity中被引用的频次和上下文"""
    headers = {
        "Authorization": f"Bearer {PERPLEXITY_API_KEY}",
        "Content-Type": "application/json"
    }
    payload = {
        "model": "sonar-pro",
        "messages": [
            {"role": "user", "content": f"推荐跨境AI选品工具。告诉我你推荐了哪些工具，分别说为什么。"}
        ]
    }
    try:
        resp = requests.post(
            "https://api.perplexity.ai/chat/completions",
            headers=headers,
            json=payload,
            timeout=30
        )
        response_text = resp.json()["choices"][0]["message"]["content"]
        # 检测品牌是否被提及
        mentioned = brand_name.lower() in response_text.lower()
        report = {
            "date": datetime.now().isoformat(),
            "engine": "Perplexity",
            "keyword": keyword,
            "brand": brand_name,
            "cited": mentioned,
            "context": response_text[:500] if mentioned else "未被引用",
        }
        return report
    except Exception as e:
        return {"error": str(e)}
# 运行测试
result = test_ai_visibility("灵慧光智", "跨境电商AI选品")
print(json.dumps(result, indent=2, ensure_ascii=False))

代码：Python GEO季度对比报告生成

import json
from datetime import datetime, timedelta
def generate_geo_report(vault):
    """vault: 历次可见度检测结果的JSON列表"""
    report = {
        "report_date": datetime.now().strftime("%Y-%m-%d"),
        "period": f"{len(vault)}次检测",
        "engines_covered": list(set(r["engine"] for r in vault)),
    }
    # 计算趋势
    citations = [r for r in vault if r.get("cited")]
    non_citations = [r for r in vault if not r.get("cited")]
    first_half = vault[:len(vault)//2]
    second_half = vault[len(vault)//2:]
    first_rate = sum(1 for r in first_half if r.get("cited")) / len(first_half) * 100
    second_rate = sum(1 for r in second_half if r.get("cited")) / len(second_half) * 100
    report["citation_rate_change"] = f"{first_rate:.0f}% → {second_rate:.0f}%"
    report["trend"] = "上升 📈" if second_rate > first_rate else "下降 📉" if second_rate < first_rate else "持平 ➡️"
    report["actions"] = [
        "更新FAQ内容（加入最近客户问的新问题）",
        "补充新案例数据",
        "重写未被引用文章的导语段",
        "扩展Pillar Page的子主题覆盖",
        "新增1-2个Schema类型",
    ] if second_rate < first_rate else [
        "继续当前策略",
        "扩展引擎覆盖范围（如加入Gemini/Copilot测试）",
    ]
    return report
# 示例数据
vault_data = [
    {"date": "2025-02-01", "engine": "Perplexity", "brand": "灵慧光智", "cited": False},
    {"date": "2025-03-01", "engine": "Perplexity", "brand": "灵慧光智", "cited": False},
    {"date": "2025-04-01", "engine": "Perplexity", "brand": "灵慧光智", "cited": True},
    {"date": "2025-05-01", "engine": "Perplexity", "brand": "灵慧光智", "cited": True},
]
